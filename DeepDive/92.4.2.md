```
 ├── bin
 │   └── MyApp   # binary executable
 │   └── sys.so  # Julia sysimage
 ├── lib
 │   ├── julia
 │   │   ├── libLLVM.so
... ... ...
 │   │   └── libz.so
 │   ├── libjulia.so -> libjulia.so.1.6
 │   ├── libjulia.so.1 -> libjulia.so.1.6
 │   └── libjulia.so.1.6
 └── share
     └── julia
         └── artifacts
             ├── 0b92c3ca1de2e9dc46a381553278edd25af54980
             ├── 56539f59fc448ad7324cfcb69239ec6e010e340e
            ...
```
